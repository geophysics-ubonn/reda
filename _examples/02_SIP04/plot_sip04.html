plot the spectrum

from reda.eis.plots import sip_response

spectrum = sip_response(
    frequencies=sip.data['frequency'].values,
    rcomplex=sip.data['zt'].values,
)

fig = spectrum.plot(filename='spectrum.pdf', dtype='r', return_fig=True)
